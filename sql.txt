-- 1. 기존에 잘못 만들어진 테이블이나 시퀀스가 있다면 삭제 (처음 만드는 거라면 생략 가능)
DROP SEQUENCE SEQ_RESERVATION;
DROP TABLE RESERVATION;

-- 2. 예약 번호를 자동으로 1씩 증가시켜 줄 시퀀스 생성
CREATE SEQUENCE SEQ_RESERVATION
START WITH 1
INCREMENT BY 1
NOCACHE;

-- 3. 예약 데이터를 담을 실제 테이블 생성
CREATE TABLE RESERVATION (
    RES_NO NUMBER PRIMARY KEY,             -- 예약 고유 번호 (PK)
    EMP_NO NUMBER NOT NULL,                -- 예약자 사원 번호
    ROOM_CODE VARCHAR2(50) NOT NULL,       -- 회의실 코드 (방 번호)
    RES_DATE DATE NOT NULL,                -- 예약 날짜
    START_TIME VARCHAR2(20) NOT NULL,      -- 시작 시간
    END_TIME VARCHAR2(20) NOT NULL,        -- 종료 시간
    PURPOSE VARCHAR2(200),                 -- 사용 목적
    STATUS VARCHAR2(20) DEFAULT '예약완료' -- 예약 상태
);

-- 4. 변경사항을 DB에 완전히 적용
COMMIT;


-- 1. 오피스 정보 테이블 생성
CREATE TABLE ROOM (
    ROOM_CODE VARCHAR2(50) PRIMARY KEY,    -- 방 코드 (예: 401, 402)
    ROOM_NAME VARCHAR2(100) NOT NULL,      -- 방 이름 (예: 대회의실)
    CAPACITY NUMBER DEFAULT 0,             -- 수용 인원
    HAS_BEAM CHAR(1) DEFAULT 'N',          -- 빔 프로젝터 유무 (Y 또는 N)
    DESCRIPTION VARCHAR2(500)              -- 방 상세 설명
);
