1) 비품 마스터 테이블 (EQUIPMENT)
사내에서 대여 가능한 모든 비품의 리스트를 관리합니다.

SQL
CREATE TABLE EQUIPMENT (
    EQ_NO NUMBER PRIMARY KEY,          -- 비품 번호 (PK)
    EQ_NAME VARCHAR2(100) NOT NULL,    -- 비품명 (예: MacBook Pro)
    TOTAL_COUNT NUMBER DEFAULT 0,      -- 총 수량
    REMAIN_COUNT NUMBER DEFAULT 0      -- 현재 대여 가능한 남은 수량
);

-- 시퀀스 생성
CREATE SEQUENCE SEQ_EQUIPMENT START WITH 1 INCREMENT BY 1;

2) 비품 대여 이력 테이블 (RENTAL_HISTORY)
누가, 언제, 어떤 비품을 빌려갔는지 기록하며, 전자 결재의 '승인 상태'를 포함합니다.

SQL
CREATE TABLE RENTAL_HISTORY (
    RENTAL_NO NUMBER PRIMARY KEY,      -- 이력 번호 (PK)
    EMP_NO NUMBER,                     -- 사원 번호 (FK)
    EQ_NO NUMBER,                      -- 비품 번호 (FK)
    RENTAL_DATE DATE DEFAULT SYSDATE,  -- 대여 신청일
    RETURN_DATE DATE,                  -- 실제 반납일
    -- 상태: 승인대기, 대여중, 반납완료, 반려됨
    STATUS VARCHAR2(50) DEFAULT '승인대기', 
    APPROVAL_STEP NUMBER DEFAULT 1,    -- 현재 결재 단계 (1~5단계)
    CONSTRAINT FK_RENTAL_EMP FOREIGN KEY (EMP_NO) REFERENCES EMPLOYEE(EMP_NO),
    CONSTRAINT FK_RENTAL_EQ FOREIGN KEY (EQ_NO) REFERENCES EQUIPMENT(EQ_NO)
);

-- 시퀀스 생성
CREATE SEQUENCE SEQ_RENTAL START WITH 1 INCREMENT BY 1;

## 1. 오라클 DB 테이블 및 시퀀스 생성
오라클 SQL Developer를 열고 아래 쿼리를 실행하여 테이블과 테스트용 데이터를 생성해 주세요.

1) 비품 마스터 테이블 (EQUIPMENT)
사내에서 관리하는 비품 목록과 수량을 저장합니다.

SQL
-- 테이블 생성
CREATE TABLE EQUIPMENT (
    EQ_NO NUMBER PRIMARY KEY,          -- 비품 번호
    EQ_NAME VARCHAR2(100) NOT NULL,    -- 비품명
    TOTAL_COUNT NUMBER DEFAULT 0,      -- 총 수량
    REMAIN_COUNT NUMBER DEFAULT 0      -- 대여 가능한 남은 수량
);

-- 시퀀스 생성
CREATE SEQUENCE SEQ_EQUIPMENT START WITH 1 INCREMENT BY 1;

2) 비품 대여 이력 테이블 (RENTAL_HISTORY)

CREATE TABLE RENTAL_HISTORY (

);
